1.2 Variables cuantitativas

histpuntajes <- function(puntajes) {
  h <- hist(puntajes, plot = FALSE)
  plot_ly(x = ~puntajes, type = "histogram", nbinsx = 2*length(h$counts), marker = list(color="lightblue1",line = list(color = "white", width = 1)))
}
